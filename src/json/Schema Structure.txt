Schema Structure

1. Users collection:
	{
		userId: Unique Identifier (autogenerated),
		name: string,
		email: string,
		password (hashed): string,
		gender: string,
		createdAt: DateTime,
		updatedAt: DateTime,
		profilePictureURL: string,
		phoneNumber?: string,
		bio?: string,
		eventPins: eventid[],
		calendarData?: [
			{
				"eventName": string,
				"start": {
					"dateTime": string,
					"timeZone": string
				},
				"end": {
					"dateTime": string,
					"timeZone": string
				},
			}
		]
	}

2. Events:
	{
		eventId: Unique Identifier,
		eventName: string,
		eventStartTime: Date,
		eventEndTime: Date,
		eventPrice: number,
		eventType: string typeof Event Type name,
		eventURL: string,
		isVirtual: boolean,
		extraEventCategories?: typeof Event Categories[], // this is if default categories associated with an event type don't include all the categories for this particular event
		eventDescription?: string,
		eventLocation?: {
			longitude: string,
			latitude: string,
			address: string,
		},
		organizerName?: string,
		eventImageURL?: string,
		eventCapacity?: number,
	}

3. Event Categories:
	{
		eventCategoryId: Unique Identifier,
		eventCategory: string,
		description: string
	}

4. Event Types:
	{
	  eventTypeId: Unique Identifier
	  name: string,
	  description: string,
	  categories: string[] typeof Event categories
	}

5. Bookings:
	{
		bookingId: Unique Identifier
		userId: number,
		eventId: number,
		reviewRating?: number,
		reviewMessage?: string
	}
